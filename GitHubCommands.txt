2. git init            #가장 먼저 git 저장소로 선언 (초기화)

3. [System.IO.File]::WriteAllLines("$PWD\.gitignore", @(".venv/", "__pycache__/", "*.pyc", ".streamlit/secrets.toml"))
	# .gitignore 작성*** -> .venv, __pycache__ 등을 미리 차단

	% 정책오류시 : Set-ExecutionPolicy -Scope CurrentUser -ExecutionPolicy RemoteSigned     # User 단위
		cf. Set-ExecutionPolicy -Scope Process -ExecutionPolicy RemoteSigned     # Terminal 단위

5. (Optional) python.exe -m pip install --upgrade pip     # pip update

6. pip install streamlit anthropic google-genai pandas numpy plotly openpyxl scipy python-dotenv streamlit-quill
	# 필수 패키지 설치 (추가/제외 가능)

7. pip freeze > requirements.txt     # 재현가능 기록관리

8. Local Pjt Folder 코드 작성 (예: stlxx.py)

9. git add .

10. git status     # add commit 대상 리스트 (.venv, secrets.toml 등은 업어야)

11. git commit -m "Initial commit"   # ""안은 commit history

12. Web에서 GitHub에 빈 저장소 생성       # README/gitignore 체크 해제!

13. git branch -M main     ->    git remote add origin https://github.com/light-dhk/xAI.git     # branch는 pointer, 주소는 예시

14. git push -u origin main     # origin은 '13'에서 정의, main은 push 대상 local branch, -u : --set-upstream : local main <-> origin/main

% 1번 PC ↔ 2번 PC Git 작업 흐름 요약
	. 1번 PC 작업 종료 : git add . -> git commit -m "..." -> git push origin main
	. 2번 PC 작업 시작 : git pull origin main
	. 2번 PC 작업 종료 : git add . -> git commit -m "..." -> git push origin main
	. 1번 PC 작업 시작 : git pull origin main
	...

			git clone <repository-URL> [폴더명(선택)]     # Git to Local copy, ex. git clone https://github.com/사용자명/my-streamlit-app.git my-app-local
			git pull     # 원격 변경사항만 가져옴
			git fetch     # 변경정보만 가져옴
			git status     #push전 변경사항 확인
			git log     # commit history check

2. Repository 생성 : Web 에서 작업 권장

	a. 방법A : 로컬폴더 없이 GitHub 생성 후 clone
		git repo create my-streamlit-app --public --clone
		cd my-streamlit-app

	b. 방법B : 로컬폴더 새 repo 연결
		mkdir my-app     # vscode root folder에서
		cd my-existing-projetc

		git init     #현재폴더 Git 저장소로 초기화
		git add .     #현재폴더 변경사항을 Staging Area로 이동
		git commit -m "Initial commit"     #변경사항을 로컬에 스냅샷 기록

		gh repo create my-streamlit-app --public --source=. --remote=origin --push     # 'repo생성 + 현재폴더 origin으로 연결 + push' 통합
			% 분할
				gh repo create my-streamlit-app --public
				git remote add origin htts://github.com/내-계정명/my-streamlit-app.git     # GitHub Repository Connect
				git remote -v     #저장소 목록 확인, -v(verbose) : URL포
				git branch -M main     #브랜치명 강제설정, -M : --move --force
				git push -u origin main     # commit을 서버(origin) main branch로 전송 (2번째 부터는 git push/pull 만 사용가능)

				% 옵션 설명
					--public / --private : 공개/비공개 설정
					--source=. : 현재 디렉토리를 소스로 사용
					--remote=origin : 원격 이름 지정
					--push : 생성 후 바로 push
					--description "설명" : repo 설명 추가 가능

3. GitHub Repository 삭제 및 재설치
	gh repo delete my-username/my-streamlit-app --yes     # --yes : 확인 프롬프트 없음
		% gh auth refresh -h github.com -s delete_repo     # delete_repo scope (삭제권한)이 없을 시, scope 추가 방법
	gh auth login     # 로그인
		% gh auth status     #로그인 상태 확인
